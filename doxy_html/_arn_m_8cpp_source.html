599 ArnLink* ArnM::linkMain( const QString& path, Arn::LinkFlags flags, Arn::ObjectSyncMode syncMode)
600 {
601  // qDebug() << "### link-main: path=" << path;
602  QString pathNorm = Arn::fullPath( path);
603  if (pathNorm.endsWith("/")) {
604  flags.set( flags.Folder);
605  pathNorm.resize( pathNorm.size() - 1); // Remove '/' at end (Also root become "")
606  }
607 
608  ArnLink* currentLink = root();
609  QStringList pathlist = pathNorm.split("/", QString::KeepEmptyParts);
610  QString growPath = "/";
611  int pathListSize = pathlist.size();
612 
613  for (int i = 1; i < pathListSize; ++i) {
614  Arn::LinkFlags subFlags;
615  subFlags.f = flags.f | flags.flagIf( i < pathListSize - 1, flags.Folder);
616  subFlags.set( flags.LastLink, i == pathListSize - 1);
617  QString subPath = pathlist.at(i);
618 
619  growPath += subPath;
620  if (subFlags.is( subFlags.Folder))
621  growPath += "/";
622 
623  currentLink = ArnM::linkMain( growPath, currentLink, subPath, subFlags, syncMode);
624  if (currentLink == 0) {
625  return 0;
626  }
627  }
628 
629  currentLink->ref();
630  return currentLink;
631 }

693  Arn::ObjectSyncMode syncMode, Arn::LinkFlags flags)
694 {

696  return 0;
697  }

699  if (!link->twinLink()) { // This link has no twin, create one

733 ArnLink* ArnM::getRawLink( ArnLink *parent, const QString& name, Arn::LinkFlags flags)
734 {
735  bool showErrors = !flags.is( flags.SilentError);
736  if (!parent) { // No parent (folder) error
737  if (showErrors) {
738  errorLog( QString(tr("Can't handle SubItem:")) + name,
739  ArnError::FolderNotOpen);
740  }
741  return 0;
742  }
743  if (parent->isRetired()) {
744  if (showErrors) {
745  errorLog( QString(tr("parent:")) + parent->linkPath(),
746  ArnError::Retired);
747  }
748  return 0;
749  }
750 
751  ArnLink *child = parent->findLink( name);
752 
753  if (child == 0) { // link not existing, create it ?
754  if (!flags.is( flags.CreateAllowed)) {
755  // Creating new items are not allowed
756  if (showErrors) {
757  errorLog( QString(tr("Path:")) + parent->linkPath() +
758  QString(tr(" Item:")) + name,
759  ArnError::NotFound);
760  }
761  return 0;
762  }
763  if (name.isEmpty() && !flags.is( flags.Folder)) {
764  // Empty names only allowed for folders
765  if (showErrors) {
766  errorLog( QString(tr("Empty leaf name, Path:")) + parent->linkPath(),
767  ArnError::CreateError);
768  }
769  return 0;
770  }
771  if (name.endsWith("!!")) {
772  // Invalid, must not have double '!' at end
773  if (showErrors) {
774  errorLog( QString(tr("Invalid name, Path:")) + parent->linkPath(),
775  ArnError::CreateError);
776  }
777  return 0;
778  }
779  // Create folders or items when needed
780  child = new ArnLink( parent, name, flags);
781  if (flags.is( flags.Folder))
782  ++_countFolder;
783  else
784  ++_countLeaf;
785  }
786  else {
787  if (child->isRetired()) {
788  if (showErrors) {
789  errorLog( QString(tr("child:")) + child->linkPath(),
790  ArnError::Retired);
791  }
792  return 0;
793  }
794  if (child->isFolder() != flags.is( flags.Folder)) {
795  // There is already a link with this name, but it is of
796  // the wrong type. This is an error.
797  if (showErrors) {
798  if (flags.is( flags.Folder)) {
799  errorLog( QString(tr("Is not folder, Path:")) + child->linkPath(),
800  ArnError::CreateError);
801  }
802  else {
803  errorLog( QString(tr("Is folder, Path:")) + child->linkPath(),
804  ArnError::CreateError);
805  }
806  }
807  return 0;
808  }
809  }
811  if (flags.is( flags.Threaded)) {
812  if (child) {
813  child->setThreaded();
814  if (child->_twin) {
815  child->_twin->setThreaded();
816  }
817  }
818  }
819 
820  return child;
821 }

863 void ArnM::destroyLinkMain( ArnLink* link, ArnLink* startLink, bool isGlobal)
864 {
865  if (!link) return;
866  if (link->isRetired()) return; // This link is already retired
867 
869  ArnLink::RetireType rt;
870  rt = startLink->isFolder() ? rt.Tree
871  : isGlobal ? rt.LeafGlobal
872  : rt.LeafLocal;
873  ArnLink* twin = link->twinLink();
874  link->setRetired( rt);
875  if (twin)
876  twin->setRetired( rt);
877  link->ref(); // At least one ref to protect this link
878

881  ArnLink* dLink = link->children().at( i);
882  if (dLink->isRetired())
883  ++i;
884  else
885  destroyLinkMain( dLink, startLink, isGlobal);
886  }
887 
889  link->doRetired( startLink, isGlobal);
890  if (twin)
891  twin->doRetired( startLink, isGlobal);
892  link->deref(); // Remove link protection, can also trigger zero-ref event
894 }

897 void ArnM::doZeroRefLink( ArnLink* link)
898 {
899  if (!link) return;
900  link->decZeroRefs();
901  if (!link->isLastZeroRef()) return; // Link reused & more zeroRefs will come
902 
903  link->setRefCount( -1); // Mark link as fully de-referenced
904  // qDebug() << "ZeroRef: set fully deref path=" << link->linkPath();
905 
906  while (link->isRetired() &&
907  link->refCount() < 0 &&
908  link->children().size() == 0) {
909  ArnLink* parent = link->parent();
910  if (Arn::debugLinkDestroy) qDebug() << "ZeroRef: delete link path=" << link->linkPath();
911 
912  if (link->isFolder())
913  --_countFolder;
914  else {
915  --_countLeaf;
916  if (link->isBiDirMode())
917  --_countLeaf;
918  }
919  delete link; // This will also delete an existing twin
920 
921  link = parent;
922  }
