485 void ArnBasicItem::arnImport( const QByteArray& data, int ignoreSame, ArnLinkHandle& handleData)
486 {
487  if (!data.isEmpty()) {
488  if (data.at(0) < 32) { // Assume Export-code
489  switch (Arn::ExportCode::fromInt( data.at(0))) {
490  case Arn::ExportCode::Variant: { // Legacy
491  QVariant value;
492  QDataStream stream( data);
493  stream.setVersion( DATASTREAM_VER);
494  quint8 dummy; // Will get Export-code
495  stream >> dummy >> value;
496  setValue( value, ignoreSame); // ArnLinkHandle not supported for QVariant
497  return;
498  }
499  case Arn::ExportCode::VariantTxt: {
500  int sepPos = data.indexOf(':', 1);
501  Q_ASSERT(sepPos > 0);
502 
503  QByteArray typeName( data.constData() + 1, sepPos - 1);
504  int type = QMetaType::type( typeName.constData());
505  if (!type) {
506  errorLog( QString("Import unknown text type:") + typeName.constData(),
507  ArnError::Undef);
508  return;
509  }
510  QVariant value( QString::fromUtf8( data.constData() + sepPos + 1,
511  data.size() - sepPos - 1));
512  if (!value.convert( QVariant::Type( type))) {
513  errorLog( QString("Can't' import data type:") + typeName.constData(),
514  ArnError::Undef);
515  return;
516  }
517 
518  setValue( value, ignoreSame); // ArnLinkHandle not supported for QVariant
519  return;
520  }
521  case Arn::ExportCode::VariantBin: {
522  if ((data.size() < 3) || (data.at(2) != DATASTREAM_VER)) {
523  errorLog( QString("Import not same DataStream version"),
524  ArnError::Undef);
525  return;
526  }
527  int sepPos = data.indexOf(':', 3);
528  Q_ASSERT(sepPos > 0);
529 
530  QByteArray typeName( data.constData() + 3, sepPos - 3);
531  int type = QMetaType::type( typeName.constData());
532  if (!type) {
533  errorLog( QString("Import unknown binary type:") + typeName.constData(),
534  ArnError::Undef);
535  return;
536  }
537 #if QT_VERSION >= 0x050000
538  void* valData = QMetaType::create( type);
539 #else
540  void* valData = QMetaType::construct( type);
541 #endif
542  Q_ASSERT( valData);
543  QDataStream stream( data);
544  stream.setVersion( DATASTREAM_VER);
545  stream.skipRawData( sepPos + 1);
546  if (!QMetaType::load( stream, type, valData)) {
547  errorLog( QString("Can't' import binary type:") + typeName.constData(),
548  ArnError::Undef);
549  QMetaType::destroy( type, valData);
550  return;
551  }
552 
553  QVariant value( type, valData);
554  QMetaType::destroy( type, valData);
555 
556  setValue( value, ignoreSame); // ArnLinkHandle not supported for QVariant
557  return;
558  }
559  case Arn::ExportCode::ByteArray:
560  setValue( data.mid(1), ignoreSame, handleData);
561  return;
562  case Arn::ExportCode::String:
563  handleData.flags().set( ArnLinkHandle::Flags::Text);
564  setValue( data.mid(1), ignoreSame, handleData);
565  //setValue( QString::fromUtf8( data.constData() + 1, data.size() - 1),
566  // ignoreSame, handleData);
567  return;
568  default: // Not supported code
569  return;
570  }
571  }
572  }
573  // Normal printable data
574  handleData.flags().set( ArnLinkHandle::Flags::Text);
575  setValue( data, ignoreSame, handleData);
576 }

579 QByteArray ArnBasicItem::arnExport() const
580 {
581  if (!_link) return QByteArray();
582 
583  QByteArray retVal;
584  Arn::DataType arnType = _link->type();
585 
586  if (arnType == Arn::DataType::Variant) {
587  QVariant value = toVariant();
588  QByteArray typeName( value.typeName());
589  int type = QMetaType::type( typeName.constData());
590  if (!type) {
591  errorLog( QString("Export unknown type:") + typeName.constData(),
592  ArnError::Undef);
593  return QByteArray();
594  }
595 
596  if (value.canConvert( QVariant::String)) { // Textual Variant
597  retVal += char( Arn::ExportCode::VariantTxt);
598  retVal += typeName;
599  retVal += ':';
600  retVal += value.toString().toUtf8();
601  }
602  else { // Binary Variant
603  QDataStream stream( &retVal, QIODevice::WriteOnly);
604  stream.setVersion( DATASTREAM_VER);
605  stream << quint8( Arn::ExportCode::VariantBin);
606  stream << quint8(0); // Spare
607  stream << quint8( DATASTREAM_VER);
608  stream.writeRawData( typeName.constData(), typeName.size());
609  stream << quint8(':');
610  if (!QMetaType::save( stream, type, value.constData())) {
611  errorLog( QString("Can't export type:") + typeName,
612  ArnError::Undef);
613  return QByteArray();
614  }
615  }
616  }
617  else if (arnType == Arn::DataType::ByteArray) {
618  retVal = char( Arn::ExportCode::ByteArray) + toByteArray();
619  }
620  else { // Expect only normal printable (could also be \n etc)
621  retVal = toString().toUtf8();
622  if (!retVal.isEmpty()) {
623  if (retVal.at(0) < 32) { // Starting char conflicting with Export-code
624  retVal.insert( 0, char( Arn::ExportCode::String)); // Stuff String-code at pos 0
625  }
626  }
627  }
628 
629  return retVal;
630 }

783 void ArnBasicItem::setValue( const ArnBasicItem& other, int ignoreSame)
784 {
785  ArnLink *link = other._link;
786 
787  if (link) {
788  switch (link->type()) {
789  case Arn::DataType::Int:
790  this->setValue( link->toInt(), ignoreSame);
791  break;
792  case Arn::DataType::Real:
793  this->setValue( link->toReal(), ignoreSame);
794  break;
795  case Arn::DataType::String:
796  this->setValue( link->toString(), ignoreSame);
797  break;
798  case Arn::DataType::ByteArray:
799  this->setValue( link->toByteArray(), ignoreSame);
800  break;
801  case Arn::DataType::Variant:
802  this->setValue( link->toVariant(), ignoreSame);
803  break;
804  case Arn::DataType::Null:
805  //cerr << "Attempt to assign null value from "
806  // << other._link->linkPath() << " to "
807  // << this->_link->linkPath() << endl;
808  errorLog( QString("Assigning Null"),
809  ArnError::ItemNotSet);
810  break;
811  default:
812  // cerr << "Unknown type when assigning " << other._link->linkPath()
813  // << " to " << this->_link->linkPath() << endl;
814  errorLog( QString("Assigning unknown type"),
815  ArnError::ItemNotSet);
816  }
817  }
818  else {
819  errorLog( QString("Assigning"),
820  ArnError::ItemNotOpen);
821  }
822 }

825 void ArnBasicItem::setValue( int value, int ignoreSame)
826 {
827  Q_D(ArnBasicItem);
828 
829  bool isIgnoreSame = (ignoreSame < 0) ? isIgnoreSameValue() : (ignoreSame != 0);
830  if (_link) {
831  if (isIgnoreSame) {
832  ArnLink* holderLink = _link->holderLink( d->_useForceKeep);
833  bool isOk;
834  if ((value == holderLink->toInt( &isOk)) && isOk) {
835  return;
836  }
837  }
838  _link->setValue( value, d->_id, d->_useForceKeep);
839  }
840  else {
841  errorLog( QString("Assigning int:") + QString::number( value),
