271 void ArnItemNet::arnEvent( QEvent* ev, bool isAlienThread)
272 {
273  ArnBasicItem::arnEvent( ev, isAlienThread);
274 
277  if (!_isMonitor) return;
278 
279  int evIdx = ev->type() - ArnEvent::baseType();
280  switch (evIdx) {
281  case ArnEvent::Idx::LinkCreate:
282  {
283  ArnEvLinkCreate* e = static_cast<ArnEvLinkCreate*>( ev);
284  if (e->isLastLink()) {
285  // qDebug() << "ArnItemNet Mon create: path=" << e->path() << " inPath=" << path();
286  sendNewItemMonEvent( e->path());
287  }
288  break;
289  }
290  case ArnEvent::Idx::ModeChange:
291  {
292  ArnEvModeChange* e = static_cast<ArnEvModeChange*>( ev);
293  Arn::XStringMap xsm;
294  xsm.add("path", e->path()).add("mode", ObjectModeToString( e->mode()));
295  // qDebug() << "ArnItemNet Mode change: data=" << xsm.toXString() << " inPath=" << path();
296  sendMonEvent( ArnMonEventType::ItemModeChg, xsm.toXString(), true);
297  break;
298  }
299  default:;
300  }
301 }

311 void ArnItemNetEar::customEvent( QEvent* ev)
312 {
313  int evIdx = ev->type() - ArnEvent::baseType();
314  switch (evIdx) {
315  case ArnEvent::Idx::LinkCreate:
316  {
317  ArnEvLinkCreate* e = static_cast<ArnEvLinkCreate*>( ev);
318  if (e->isLastLink() && e->arnLink()->isFolder()) {
319  // qDebug() << "ArnItemNetEar create tree: path=" << e->path();
320  emit arnTreeCreated( e->path());
321  }
322  break;
323  }
324  case ArnEvent::Idx::Retired:
325  {
326  ArnEvRetired* e = static_cast<ArnEvRetired*>( ev);
327  QString destroyPath = e->isBelow() ? e->startLink()->linkPath() : path();
328  if (Arn::debugLinkDestroy) qDebug() << "ArnItemNetEar retired: path=" << destroyPath
329  << " isGlobal=" << e->isGlobal()
330  << "inPath=" << path();
331  if (e->startLink()->isFolder()) {
332  emit arnTreeDestroyed( destroyPath, e->isGlobal());
333  }
334  break;
335  }
336  default:;
337  }
338 
339  return ArnItem::customEvent( ev);
