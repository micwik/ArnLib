61 int ArnEvent::baseType( int setVal)
62 {
63  static int bt = -1;
64 
65  if (bt > 0) return bt;
66 
68  bt = setVal;
69  if (bt < 0)
70  bt = 2022; // Arn default
71  Q_ASSERT_X((bt >= QEvent::User) && (bt + Idx::N <= QEvent::MaxUser), "ArnEvent::baseType()",
72  "Selected base for ArnEvent number is out of boundary for Qt User event");
73 
75  for (int i = 0; i < Idx::N; ++i) {
76  int wantType = bt + i;
77  int gotType = QEvent::registerEventType( wantType);
78  Q_ASSERT_X(gotType == wantType, "ArnEvent::baseType()",
79  "Assigning event number for ArnEvent is already taken,"
80  " use ArnEvent::baseType() to set other base event number for ArnEvents.");
81  }
82 
83  return bt;
84 }

142 void ArnEvent::setTargetPendingChain( ArnEvent** targetPendingChain)
143 {
144  if (_targetMutex) _targetMutex->lock();
145 
146  _targetPendingChain = targetPendingChain;
147 
148  if (_targetPendingChain) { // If pending chain used, link this event to it
149  _targetNextPending = *_targetPendingChain;
150  *_targetPendingChain = this;
151  if (_targetNextPending)
152  _targetNextPending->_targetPendingChain = &_targetNextPending;
153  }
154 
155  if (_targetMutex) _targetMutex->unlock();
156 }

165 void ArnEvent::inhibitPendingChain()
166 {
167  QMutex* targetMutex = _targetMutex;
168  if (targetMutex) targetMutex->lock();
169 
170  if (_targetPendingChain) { // If pending chain used, inhibit all events in it
171  ArnEvent* ev = this;
172  do {
173  ArnEvent* evNext = ev->_targetNextPending;
174  ev->_target = 0; // Inhibit event, i.e. it will be dropped when delivered
175  ev->_targetMutex = 0;
176  *ev->_targetPendingChain = 0;
177  ev->_targetNextPending = 0;
178  ev = evNext;
179  } while (ev);
180  }
181 
182  if (targetMutex) targetMutex->unlock();
183 }
