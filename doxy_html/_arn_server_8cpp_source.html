216 void ArnServer::start( int port, QHostAddress listenAddr)
217 {
218  Q_D(ArnServer);
219 
220  if (port < 0) {
221  switch (d->_serverType) {
222  case Type::NetSync:
223  port = Arn::defaultTcpPort;
224  break;
225  default:
226  ArnM::errorLog( QString(tr("Unknown Arn server Type:")) + QString::number( d->_serverType),
227  ArnError::Undef);
228  return;
229  }
230  }
231 
232  if (d->_tcpServer->listen( listenAddr, port)) {
233  d->_tcpServerActive = true;
234 
235  connect( d->_tcpServer, SIGNAL(newConnection()), this, SLOT(tcpConnection()));
236  }
237  else {
238  ArnM::errorLog( QString(tr("Failed start Arn Server Port:")) + QString::number( port),
239  ArnError::ConnectionError);
240  }
241 }

301 bool ArnServer::isDemandLoginNet( const QHostAddress& remoteAddr) const
302 {
303  Q_D(const ArnServer);
304 
305  foreach (const QString& noLoginNet, d->_noLoginNets) {
306  bool chkLocalHost = noLoginNet == "localhost";
307  if (chkLocalHost || (noLoginNet == "localnet")) {
308  if ((remoteAddr == QHostAddress( QHostAddress::LocalHost))
309  || (remoteAddr == QHostAddress( QHostAddress::LocalHostIPv6)))
310  return false; // Localhost, ok for both localhost & localnet
311 
312  foreach (QNetworkInterface interface, QNetworkInterface::allInterfaces()) {
313  QNetworkInterface::InterfaceFlags flags = interface.flags();
314  if (flags.testFlag( QNetworkInterface::IsPointToPoint)
315  || flags.testFlag( QNetworkInterface::IsLoopBack))
316  continue;
317 
318  foreach (QNetworkAddressEntry entry, interface.addressEntries()) {
319  QAbstractSocket::NetworkLayerProtocol prot = entry.ip().protocol();
320  if ((prot != QAbstractSocket::IPv4Protocol) && (prot != QAbstractSocket::IPv6Protocol))
321  continue;
322 
323  if (entry.ip() == remoteAddr) // Address to this host ip, ok for both localhost & localnet
324  return false;
325 
326  QString subNetString = entry.ip().toString() + "/" + entry.netmask().toString();
327  if (!chkLocalHost && remoteAddr.isInSubnet( QHostAddress::parseSubnet( subNetString)))
328  return false;
329  }
330  }
331  }
332  else if (noLoginNet == "any") {
333  return false;
334  }
335  else {
336  if (remoteAddr.isInSubnet( QHostAddress::parseSubnet( noLoginNet)))
337  return false;
338  }
339  }
340 
341  return true;
342 }
