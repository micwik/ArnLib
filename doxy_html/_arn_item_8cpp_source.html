440 void ArnItem::itemUpdated( const ArnLinkHandle& handleData, const QByteArray* value)
441 {
442  Q_D(ArnItem);
443 
444  if (!value) { // Update of item with no data supplied
445  if (d->_delayTimer) {
446  if (!d->_delayTimer->isActive()) {
447  d->_delayTimer->start( this);
448  // qDebug() << "ArnItem delay start: path=" << path();
449  }
450  }
451  else {
452  doItemUpdate( handleData);
453  }
454  }
455  else { // Update of item with data supplied (pipe in multi-thread)
456  if (d->_emitChanged) {
457  emit changed();
458  }
459  if (d->_emitChangedInt) {
460  emit changed( int( value->toInt()));
461  }
462  if (d->_emitChangedReal) {
463 #if defined( ARNREAL_FLOAT)
464  emit changed( float( value->toFloat()));
465 #else
466  emit changed( double( value->toDouble()));
467 #endif
468  }
469  if (d->_emitChangedBool) {
470  emit changed( bool( value->toInt() != 0));
471  }
472  if (d->_emitChangedString) {
473  emit changed( QString::fromUtf8( value->constData(), value->size()));
474  }
475  if (d->_emitChangedByteArray) {
476  emit changed( *value);
477  }
478  if (d->_emitChangedVariant) {
479  // Can only handle printable value ...
480  emit changed( QVariant( QString::fromUtf8( value->constData(), value->size())));
481  }
482  resetOnlyEcho(); // Nothing else yet ...
483  }
484 }

488 {
489  Q_UNUSED(handleData);
