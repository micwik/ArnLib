135 void ArnPipe::itemUpdated( const ArnLinkHandle& handleData, const QByteArray* value)
136 {
137  ArnItemB::itemUpdated( handleData, value);
138 
139  if (_useCheckSeq && handleData.has( ArnLinkHandle::SeqNo)) {
140  int seqNum = handleData.valueRef( ArnLinkHandle::SeqNo).toInt();
141  if (seqNum != _checkSeqNum) { // Sequence not matching
142  if (_checkSeqNum != -1) // If not initial, this is out of sequence
143  emit outOfSequence();
144  _checkSeqNum = seqNum; // Resync to this received SeqNum
145  }
146  _checkSeqNum = (_checkSeqNum + 1) % 1000;
147  }
148  if (value)
149  emit changed( *value);
150  else
151  emit changed( toByteArray());
152 }
