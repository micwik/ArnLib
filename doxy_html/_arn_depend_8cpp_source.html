237 void ArnDepend::startMonitor()
238 {
239  Q_D(ArnDepend);
240 
241  if (d->_started) return; // Already started
242 
243  // Connect & check all depency slots
244  foreach( DepSlot* slot, d->_depTab) {
245  connect( &slot->arnEchoPipe, SIGNAL(changed(QString)), this, SLOT(echoCheck(QString)));
246  echoCheck("", slot);
247  }
248  d->_started = true;
249 }

265 void ArnDepend::echoCheck( const QString& echo, DepSlot* slot)
266 {
267  Q_D(ArnDepend);
268 
269  if (Arn::debugDepend) qDebug() << "echoCheck: monitorName=" << d->_name;
270  if (slot == 0) {
271  ArnItem* arnItem = qobject_cast<ArnItem*>( sender());
272  if (arnItem) slot = static_cast<DepSlot*>( arnItem->reference());
273  }
274  if (slot == 0) {
275  ArnM::errorLog( QString(tr("Can't get slot for DependEchoCheck monitor=")) + d->_name,
276  ArnError::Undef);
277  return;
278  }
279  if (slot->isEchoOk) return; // Echo already ok, test just in case ...
280 
281  if (echo.isEmpty()) {
282  if (Arn::debugDepend) qDebug() << "echoCheck: Send request monitorName=" << d->_name << " req=" << d->_uuid;
283  slot->arnEchoPipe = d->_uuid; // Dependency request
284  }
285  else {
286  d->_timerEchoRefresh->start(); // Only when getting echo, to avoid fillup during disconnect
287  }
288  if (echo == d->_uuid) { // Dependency echo ok
289  slot->isEchoOk = true;
290  disconnect( &slot->arnEchoPipe, SIGNAL(changed(QString)), this, SLOT(echoCheck(QString))); // Avoid more signals
291  if (slot->useStateCheck) { // Go on with state check
292  connect( &slot->arnStateId, SIGNAL(changed()), this, SLOT(stateCheck()));
293  connect( &slot->arnStateName, SIGNAL(changed()), this, SLOT(stateCheck()));
294  stateCheck( slot);
295  }
296  else { // This dependency is ok
297  doDepOk( slot);
298  }
299  }
300 }

303 void ArnDepend::stateCheck( DepSlot* slot)
304 {
305  Q_D(ArnDepend);
306 
307  if (slot == 0) {
308  ArnItem* arnItem = qobject_cast<ArnItem*>( sender());
309  if (arnItem) slot = static_cast<DepSlot*>( arnItem->reference());
310  }
311  if (slot == 0) {
312  ArnM::errorLog( QString(tr("Can't get slot for DependStateCheck monitor=")) + d->_name,
313  ArnError::Undef);
314  return;
315  }
316  if (slot->isStateOk) return; // State already ok
317 
318  if (!slot->stateName.isEmpty() && (slot->arnStateName.toString() == slot->stateName)) {
319  slot->isStateOk = true;
320  doDepOk( slot);
321  }
322  else if ((slot->stateId > 0) && (slot->arnStateId.toInt() >= slot->stateId)) {
323  slot->isStateOk = true;
324  doDepOk( slot);
325  }
326 }

341 void ArnDepend::deleteSlot( void* slot_)
342 {
343  Q_D(ArnDepend);
344 
345  DepSlot* slot = reinterpret_cast<DepSlot*>( slot_);
346  if (Arn::debugDepend) qDebug() << "deleteSlot monitorName=" << d->_name;
347  if (!d->_depTab.removeOne( slot)) {
348  ArnM::errorLog( QString(tr("Can't get slot for delete Depend monitor=")) + d->_name,
349  ArnError::Undef);
350  return;
351  }
352  delete slot;
353 
354  if (d->_depTab.isEmpty()) { // All dependencys ok
