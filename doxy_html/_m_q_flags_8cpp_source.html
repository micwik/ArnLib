140 void EnumTxt::addBitSet( XStringMap& xsm, quint16 nameSpace, bool neverHumanize)
141 {
142  if (!_isFlag) return; // Only for flags
143 
144  EnumTxtKey keyStart( 0, nameSpace, _isFlag);
145  EnumTxtKey keyStop( UINT_MAX, nameSpace, _isFlag);
146  uint bitNum = 0;
147 
148  QMap<EnumTxtKey,const char*>::iterator i = _enumTxtTab.lowerBound( keyStart);
149  while (i != _enumTxtTab.end()) {
150  const EnumTxtKey& keyStored = i.key();
151  if (keyStop < keyStored) break;
152 
153  uint enumValStored = keyStored._enumVal;
154  while (uint(1 << bitNum) < enumValStored)
155  ++bitNum;
156 
157  QByteArray enumTxt;
158  if ((nameSpace > 0) || neverHumanize) { // Enum name is used as is
159  enumTxt = i.value();
160  }
161  else { // Enum name is humanized (Leading capital and separated words)
162  enumTxt = humanize( QString::fromUtf8( i.value())).toUtf8();
163  }
164  xsm.add("B" + QByteArray::number( bitNum), enumTxt);
165  ++i;
166  }
167 }

273 void EnumTxt::setMissingTxt( quint16 toNameSpace, quint16 fromNameSpace, bool neverHumanize)
274 {
275  if (toNameSpace == 0) return; // Not allowed to change original enum texts
276  if (toNameSpace == fromNameSpace) return;
277 
278  XStringMap xsmFrom;
279  XStringMap xsmTo;
280  addEnumSet( xsmFrom, fromNameSpace);
281  addEnumSet( xsmTo, toNameSpace);
282 
283  for (int i = 0; i < xsmFrom.size(); ++i) {
284  const QByteArray& enumValKey = xsmFrom.keyRef( i);
285  if (xsmTo.indexOf( enumValKey) < 0) { // Missing enumval in target
286  int enumVal = enumValKey.toInt();
287  const char* enumTxt = getTxt( enumVal, fromNameSpace); // Get the original txt ptr
288 
289  if ((fromNameSpace > 0) || neverHumanize) { // Enum name is used as is
290  setTxtRefAny( enumTxt, enumVal, toNameSpace);
291  }
292  else { // Enum name is humanized (Leading capital and separated words)
293  QByteArray enumTxtHuman = humanize( QString::fromUtf8( enumTxt)).toUtf8();
294  setTxt( enumTxtHuman.constData(), enumVal, toNameSpace);
295  }
296  }
297  }

301 QString EnumTxt::humanize( const QString& txt)
302 {
303  QString retVal;
304  bool wasDigit = false;
305  bool wasUpper = false;
306  bool wasLower = false;
307  bool wasWantSpace = false;
308  bool wasSpace = false;
309 
310  for (int i = 0; i < txt.size(); ++i) {
311  QChar c = txt.at(i);
312  bool isDigit = c.isDigit();
313  bool isUpper = c.isUpper();
314  bool isLower = c.isLower();
315  bool wantSpace = false;
316 
317  if (i > 0)
318  c = c.toLower();
319 
320  if (i == 0)
321  c = c.toUpper();
322  else if (c == '_')
323  c = ' ';
324  else if ((i >= 2) && wasUpper && isLower && !wasWantSpace)
325  retVal.insert( retVal.size() - 1, ' ');
326  else if (wasLower && isUpper)
327  wantSpace = true;
328  else if (!wasDigit && isDigit && !wasSpace)
329  wantSpace = true;
330  else if (wasDigit && !isDigit)
331  wantSpace = true;
332 
333  if (wantSpace)
334  retVal += ' ';
335 
336  retVal += c;
337 
338  wasSpace = c.isSpace();
339  wasDigit = isDigit;
340  wasUpper = isUpper;
341  wasLower = isLower;
342  wasWantSpace = wantSpace;
343  }
344 
345  return retVal;
346 }

368 void EnumTxt::setupFromMetaObject()
369 {
370  if (_metaObj.enumeratorCount() < 1) return;
371 
372  QMetaEnum metaEnum = _metaObj.enumerator(0);
373  int nKeys = metaEnum.keyCount();
374  for (int i = 0; i < nKeys; ++i) {
375  int enumVal = metaEnum.value(i);
376  if (_isFlag && !isPower2( enumVal))
377  continue; // Not a single bit for flags
378  setTxtRefAny( metaEnum.key(i), enumVal, 0);
