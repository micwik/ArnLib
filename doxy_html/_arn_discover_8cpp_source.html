453 bool ArnDiscoverBrowserB::goTowardState( int index, ArnDiscoverInfo::State state)
454 {
455  Q_D(ArnDiscoverBrowserB);
456 
457  if ((index < 0) || (index >= d->_activeServInfos.size())) return false; // Out of index
458 
459  ArnDiscoverInfo& info = d->_activeServInfos[ index];
460  if (state <= info.state()) return false; // Can only go forward
461 
462  if (info.inProgress()) { // Next state is in progress
463  info.d_ptr->_stopState = state; // Just update final state
464  }
465  else if (info.state() <= info.stopState()) { // Nothing in progress
466  info.d_ptr->_stopState = state; // Update for new final state
467  doNextState( info); // Startup state change
468  }
469  else {
470  Q_ASSERT_X(false, "ArnDiscoverBrowser::goTowardState()", "State passed stopState");
471  return false; // Internal error
472  }
473 
474  return true;
475 }

501 int ArnDiscoverBrowserB::resolve( const QString& serviceName, bool forceUpdate)
502 {
503  Q_D(ArnDiscoverBrowserB);
504 
505  if (Arn::debugDiscover) qDebug() << "Man resolve Service: name=" << serviceName;
506 
507  int id = serviceNameToId( serviceName);
508  if ((id >= 0) && forceUpdate) {
509  int index = IdToIndex( id);
510  removeServiceInfo( index);
511  id = -1; // Mark not found
512  }
513 
514  ArnDiscoverInfo* info = 0;
515  int index = -1;
516  if (id < 0) { // Not found, resolve new service
517  id = ArnZeroConfBrowser::getNextId();
518  index = newServiceInfo( id, serviceName, QString());
519  Q_ASSERT(index >= 0); // Already resolving, internal error ...
520 
521  info = &d->_activeServInfos[ index];
522  doNextState( *info);
523  }
524  else { // Already resolving / resolved
525  index = IdToIndex( id);
526  info = &d->_activeServInfos[ index];
527  if (!info->inProgress()) // Not in progress, possibly do retry error
528  doNextState( *info);
529  }
530  Q_ASSERT(info && (index >= 0));
531 
532  if (info->inProgress()) {
533  info->d_ptr->_resolvCode = ArnZeroConf::Error::Running;
534  }
535  emit infoUpdated( index, info->state());
536 
537  return index;
538 }

716 void ArnDiscoverBrowserB::doNextState( ArnDiscoverInfo& info)
717 {
718  if (info.state() >= info.stopState()) return; // At stop state, do nothing more now
719 
720  switch (info.state()) {
721  case ArnDiscoverInfo::State::HostInfoErr: // Will retry resolv
722  // Fall throu
723  case ArnDiscoverInfo::State::ServiceName:
724  {
725  info.d_ptr->_state = ArnDiscoverInfo::State::ServiceName;
726 
727  ArnZeroConfResolve* ds = new ArnZeroConfResolve( info.serviceName(), this);
728  ds->setId( info.d_ptr->_id);
729  connect( ds, SIGNAL(resolveError(int,int)), this, SLOT(onResolveError(int,int)));
730  connect( ds, SIGNAL(resolved(int,QByteArray)), this, SLOT(onResolved(int,QByteArray)));
731  ds->resolve();
732  break;
733  }
734  case ArnDiscoverInfo::State::HostIpErr: // Will retry ip lookup
735  // Fall throu
736  case ArnDiscoverInfo::State::HostInfo:
737  {
738  info.d_ptr->_state = ArnDiscoverInfo::State::HostInfo;
739 
740  ArnZeroConfLookup* ds = new ArnZeroConfLookup( info.hostName(), this);
741  ds->setId( info.d_ptr->_id);
742  connect( ds, SIGNAL(lookupError(int,int)), this, SLOT(onLookupError(int,int)));
743  connect( ds, SIGNAL(lookuped(int)), this, SLOT(onLookuped(int)));
744  if (Arn::debugDiscover) qDebug() << "LookingUp host=" << info.hostName() << " Id=" << info.d_ptr->_id;
745  ds->lookup();
746  break;
747  }
748  default:
749  break;
750  }
751 }
