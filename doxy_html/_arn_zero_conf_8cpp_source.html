107 void ArnZeroConfB::parseFullDomain( const QByteArray& domainName)
108 {
109  QRegExp rx("^((?:\\\\{2,2}|\\\\\\.|\\\\\\d{3,3}|[^\\\\\\.])+)\\.(.+\\._(?:tcp|udp))\\.(.+)");
110  if (rx.indexIn( QString::fromUtf8( domainName.constData())) != -1) {
111  setServiceType(rx.cap(2));
112  setDomain(rx.cap(3));
113 
114  QByteArray servName = rx.cap(1).toUtf8();
115  for (int i= 0; i < servName.length(); ++i) {
116  if (servName.at(i) != '\\') continue;
117 
118  char c = servName.at(i + 1);
119  if ((c == '\\') || (c == '.'))
120  servName.remove(i, 1);
121  else if (isdigit(c))
122  servName.replace(i, 4, QByteArray(1, char(servName.mid(i + 1, 3).toInt())));
123  }
124 
125  setServiceName( QString::fromUtf8( servName.constData(), servName.length()));
126  }
127 }

130 QByteArray ArnZeroConfB::escapedName( const QByteArray& name)
131 {
132  QByteArray retVal;
133  const char* namep = name.constData();
134  int len = name.size();
135 
136  for (int i = 0; i < len; ++i) {
137  uchar c = *namep++;
138  if (c <= ' ') {
139  retVal += '\\';
140  retVal += QByteArray::number(c).rightJustified(3, '0');
141  }
142  else if ((c == '.') || (c == '\\')) {
143  retVal += '\\';
144  retVal += char(c);
145  }
146  else
147  retVal += char(c);
148  }
149 
150  return retVal;
151 }

209 bool ArnZeroConfB::getTxtRecordMap( XStringMap& xsm)
210 {
211  xsm.clear();
212  QByteArray txtRec = _txtRec;
213  forever {
214  if (txtRec.isEmpty()) return true; // No more params
215  int parLen = quint8( txtRec.at(0));
216  if (parLen > (txtRec.size() - 1)) return false; // Too long parameter length
217  QByteArray par( txtRec.constData() + 1, parLen);
218  txtRec.remove(0, parLen + 1);
219 
220  int eqPos = par.indexOf('=');
221  xsm.add( par.mid(0, eqPos), par.mid( eqPos + 1));
222  }
223 }

267 void ArnZeroConfB::setServiceType( const QString& type)
268 {
269  QStringList tp = type.split('.');
270 
271  if ((tp.size() == 2) && (tp.at(0).size() > 0) && (tp.at(0)[0] == '_'))
272  _serviceType = tp.at(0).mid(1);
273  else
274  _serviceType = type;
275 
276  if (tp.size() >= 2) {
277  QString sockTypePart = tp.last();
278  if (sockTypePart == "_udp")
279  setSocketType( QAbstractSocket::UdpSocket);
280  else if (sockTypePart == "_tcp")
281  setSocketType( QAbstractSocket::TcpSocket);
282  else
283  setSocketType( QAbstractSocket::UnknownSocketType);
284  }
285  else
286  setSocketType( QAbstractSocket::TcpSocket);
287 }

422 void ArnZeroConfRegister::registerService( bool noAutoRename)
423 {
424  if (state() != ArnZeroConf::State::None) {
425  qWarning() << "ZeroConfRegister: Error register service while not in None state";
426  emit registrationError(0);
427  return;
428  }
429 
430  QByteArray txtRec = txtRecord();
431  if (txtRec.isEmpty())
432  txtRec.fill(0, 1);
433 
434  QByteArray serviceTypes = fullServiceType().toUtf8();
435  foreach (const QString& subType, _serviceSubTypes) {
436  serviceTypes += ",_" + escapedName( subType.toUtf8());
437  }
438 
439  if (Arn::debugZeroConf) qDebug() << "Register: serviceTypes=" << serviceTypes.constData() << "";
440  DNSServiceErrorType err;
441  err = DNSServiceRegister(&_sdRef,
442  noAutoRename ? kDNSServiceFlagsNoAutoRename : 0,
443  _iface,
444  serviceName().toUtf8().constData(),
445  serviceTypes.constData(),
446  domain().toUtf8().constData(),
447  host().toUtf8().constData(),
448  qToBigEndian( port()),
449  txtRec.length(),
450  txtRec.constData(),
451  ArnZeroConfIntern::registerServiceCallback,
452  this);
453  if (err != kDNSServiceErr_NoError) {
454  _state = ArnZeroConf::State::None;
455  emit registrationError(err);
456  }
457  else {
458  _state = ArnZeroConf::State::Registering;
459 #ifndef MDNS_INTERN
460  _notifier = new QSocketNotifier( DNSServiceRefSockFD( _sdRef), QSocketNotifier::Read, this);
461  connect( _notifier, SIGNAL(activated(int)), this, SLOT(socketData()));
462 #endif
463  }
464 }

572 void ArnZeroConfResolve::resolve( bool forceMulticast)
573 {
574  if (_id < 0) // No valid id set, get one
575  _id = ArnZeroConfB::getNextId();
576 
577  if (state().isAny( ArnZeroConf::State::InProgress)) {
578  qWarning() << "ZeroConfResolv: Error resolve service when operation still in progress";
579  emit resolveError( _id, ArnZeroConf::Error::BadReqSeq);
580  return;
581  }
582  releaseResolve();
583 
584  DNSServiceErrorType err;
585  err = DNSServiceResolve(&_sdRef,
586  (forceMulticast ? kDNSServiceFlagsForceMulticast : 0),
587  _iface,
588  serviceName().toUtf8().constData(),
589  fullServiceType().toUtf8().constData(),
590  domain().toUtf8().constData(),
591  ArnZeroConfIntern::resolveServiceCallback,
592  this);
593  if (err != kDNSServiceErr_NoError) {
594  _sdRef = 0;
595  _state.set( ArnZeroConf::State::Resolve, false);
596  emit resolveError( _id, err);
597  }
598  else {
599  _state.set( ArnZeroConf::State::Resolving);
600  _operationTimer->start();
601 
602 #ifndef MDNS_INTERN
603  _notifier = new QSocketNotifier( DNSServiceRefSockFD( _sdRef), QSocketNotifier::Read, this);
604  connect( _notifier, SIGNAL(activated(int)), this, SLOT(socketData()));
605 #endif
606  }
607 }

634 void DNSSD_API ArnZeroConfIntern::resolveServiceCallback(
635  DNSServiceRef sdRef, DNSServiceFlags flags, quint32 iface,
636  DNSServiceErrorType errCode, const char* fullname, const char* host, quint16 port,
637  quint16 txtLen, const unsigned char* txt, void* context)
638 {
639  Q_UNUSED(sdRef);
640  Q_UNUSED(flags);
641  ArnZeroConfResolve* self = reinterpret_cast<ArnZeroConfResolve*>(context);
642  Q_ASSERT(self);
643 
644  self->_operationTimer->stop();
645 
646  if (self->_id < 0) // No valid id set, get one
647  self->_id = ArnZeroConfB::getNextId();
648 
649  if (Arn::debugZeroConf) qDebug() << "Resolve callback errCode=" << errCode;
650  if (errCode == kDNSServiceErr_NoError) {
651  QString resHost = QString::fromUtf8( host);
652  if (resHost.endsWith('.')) // MW: Remove strangely added "."
653  resHost.resize( resHost.size() - 1);
654  self->parseFullDomain( fullname);
655  self->setHost( resHost);
656  self->setPort( qFromBigEndian( port));
657  self->setTxtRecord( txtLen > 0 ? QByteArray((const char*) txt, txtLen) : QByteArray());
658  self->_iface = iface;
659  self->_state.set( ArnZeroConf::State::Resolving, false);
660  self->_state.set( ArnZeroConf::State::Resolved);
661  emit self->resolved( self->_id, fullname);
662  }
663  else {
664  self->_state.set( ArnZeroConf::State::Resolve, false);
665  emit self->resolveError( self->_id, errCode);
666  }
667 }

723 void ArnZeroConfLookup::lookup( bool forceMulticast)
724 {
725  if (_id < 0) // No valid id set, get one
726  _id = ArnZeroConfB::getNextId();
727 
728  if (state().isAny( ArnZeroConf::State::InProgress)) {
729  qWarning() << "ZeroConfLookup: Error lookup host when operation still in progress";
730  emit lookupError( _id, ArnZeroConf::Error::BadReqSeq);
731  return;
732  }
733  releaseLookup();
734 
735  bool useQtLookup = false; // Default
736  useQtLookup |= !forceMulticast && !_host.endsWith(".local");
737  useQtLookup |= _isForceQtDnsLookup;
738 #ifndef MDNS_HAVE_LOOKUP
739  useQtLookup = true;
740 #endif
741 
742  // Qt DNS lookup
743  if (useQtLookup) {
744  int ipLookupId = QHostInfo::lookupHost( _host, this, SLOT(onIpLookup(QHostInfo)));
745  if (Arn::debugZeroConf) qDebug() << "ZeroConfLookup Qt DNS: host=" << _host << " lookupId=" << ipLookupId;
746  _state.set( ArnZeroConf::State::LookingUp);
747  return;
748  }
749 
750 #ifdef MDNS_HAVE_LOOKUP
751  // Multicast DNS lookup
752  if (Arn::debugZeroConf) qDebug() << "ZeroConfLookup mDNS: host=" << _host << " id=" << _id;
753  DNSServiceErrorType err;
754  err = DNSServiceGetAddrInfo(&_sdRef,
755  (forceMulticast ? kDNSServiceFlagsForceMulticast : 0),
756  _iface,
757  kDNSServiceProtocol_IPv4,
758  _host.toUtf8().constData(),
759  ArnZeroConfIntern::lookupHostCallback,
760  this);
761  if (err != kDNSServiceErr_NoError) {
762  _sdRef = 0;
763  _state.set( ArnZeroConf::State::Lookup, false);
764  emit lookupError( _id, err);
765  }
766  else {
767  _state.set( ArnZeroConf::State::LookingUp);
768  _operationTimer->start();
769 
770 #ifndef MDNS_INTERN
771  _notifier = new QSocketNotifier( DNSServiceRefSockFD( _sdRef), QSocketNotifier::Read, this);

995 void DNSSD_API ArnZeroConfIntern::browseServiceCallback(
996  DNSServiceRef sdRef, DNSServiceFlags flags, quint32 iface,
997  DNSServiceErrorType errCode, const char* serviceName, const char* regtype,
998  const char* replyDomain, void* context)
999 {

1001  Q_UNUSED(iface);

1003 
1004  ArnZeroConfBrowser* self = reinterpret_cast<ArnZeroConfBrowser*>(context);
1005  if (errCode == kDNSServiceErr_NoError) {
1006  bool isAdded = (flags & kDNSServiceFlagsAdd) != 0;
1007  QString servName = QString::fromUtf8( serviceName);
1008  QString repDomain = QString::fromUtf8( replyDomain);
1009  if (isAdded != self->_activeServiceNames.contains( servName)) { // Not multiple add or remove, ok
1010  int id = 0;
1011  if (isAdded) {
1012  id = ArnZeroConfB::getNextId();
1013  self->_activeServiceNames.insert( servName, id);
1014  emit self->serviceAdded( id, servName, repDomain);
1015  }
1016  else {
1017  id = self->_activeServiceNames.value( servName);
1018  self->_activeServiceNames.remove( servName);
1019  emit self->serviceRemoved( id, servName, repDomain);
1020  }
1021  emit self->serviceChanged( id, isAdded, servName, repDomain);
1022  }
1023  }
1024  else {
1025  emit self->browseError( errCode);
