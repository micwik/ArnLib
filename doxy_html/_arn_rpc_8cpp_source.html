107 int ArnDynamicSignals::qt_metacall( QMetaObject::Call call, int id, void **arguments)
108 {
109  id = QObject::qt_metacall( call, id, arguments);
110  if ((id < 0) || (call != QMetaObject::InvokeMetaMethod))
111  return id;
112 
113  Q_ASSERT(id < _slotTab.size());
114 
115  MQGenericArgument args[8];
116  const Slot& slot = _slotTab.at( id);
117  int argNum = slot.typeNames.size();
118 
119  for (int i = 0; i < argNum; ++i) {
120  args[i] = MQGenericArgument( slot.typeNames.at(i).constData(),
121  slot.parNames.at(i).constData(),
122  arguments[i + 1]);
123  }
124  _rpc->invoke( QString::fromLatin1( slot.funcName),
125  slot.invokeFlags,
126  args[0],
127  args[1],
128  args[2],
129  args[3],
130  args[4],
131  args[5],
132  args[6],
133  args[7]);
134 
135  return -1;

357 void ArnRpc::addSenderSignals( QObject* sender, QString prefix)
358 {
359  const QMetaObject* metaObject = sender->metaObject();
360  int methodCount = metaObject->methodCount();
361  QByteArray methodSignCompHead;
362  int methodIdCompHead = -1;
363  for (int methodId = 0; methodId < methodCount; ++methodId) {
364  QMetaMethod method = metaObject->method( methodId);
365  if (method.methodType() != QMetaMethod::Signal) continue;
366 
367  QByteArray methodSign = methodSignature( method);
368  if (!methodSign.startsWith( prefix.toLatin1())) continue;
369 
370  QByteArray methodName = methodSign.left( methodSign.indexOf('('));
371  QByteArray funcName = methodName.mid( prefix.size());
372 
373  QByteArray methodSignComp = methodSign;
374  methodSignComp.chop(1); // Remove last ")"
375 
376  if (!_mode.is( Mode::NoDefaultArgs) // When using Default args ...
377  && methodSignCompHead.startsWith( methodSignComp) // Starts with same signatur ...
378  && hasSameParamNames( method, metaObject->method( methodIdCompHead))) // and same param names
379  continue; // Skip it, to prohibit multiple rpc calls.
380 
381  methodSignCompHead = methodSignComp;
382  methodIdCompHead = methodId;
383  _dynamicSignals->addSignal( sender, methodId, funcName);
384  }
385 }

487 bool ArnRpc::xsmAddArg( XStringMap& xsm, const MQGenericArgument& arg, uint index, int& nArg)
488 {
489  if (!arg.name() || !arg.label() || !arg.data()) return true; // Empty arg
490  if (nArg + 1 != int( index)) return true; // Out of seq - Finished
491 
493  QByteArray typeName = arg.name();
494  int type = QMetaType::type( typeName.constData());
495  if (!type) {
496  errorLog( QString(tr("Unknown type:") + typeName.constData()),
497  ArnError::RpcInvokeError);
498  return false;
499  }
500  const RpcTypeInfo& rpcTypeInfo = typeInfoFromId( type);
501 
503  QByteArray argLabel = arg.label();
504 
506  QByteArray argDataDump;
507  QStringList argDataList;
508  bool isBinaryType = false;
509  QVariant varArg( type, arg.data());
510  if (type == QMetaType::QStringList) {
511  argDataList = varArg.toStringList();
512  }
513  else if (type == QMetaType::QByteArray) {
514  argDataDump = varArg.toByteArray();
515  }
516  else if (varArg.canConvert( QVariant::String)) {
517  argDataDump = varArg.toString().toUtf8();
518  }
519  else {
520  QDataStream stream( &argDataDump, QIODevice::WriteOnly);
521  stream.setVersion( DATASTREAM_VER);
522  stream << quint8(0); // Spare
523  stream << quint8( DATASTREAM_VER);
524  if (!QMetaType::save( stream, type, arg.data())) {
525  errorLog( QString(tr("Can't export type:") + typeName.constData()),
526  ArnError::RpcInvokeError);
527  return false;
528  }
529  isBinaryType = true;
530  }
531 
533  QByteArray argKey;
534  if (!rpcTypeInfo.typeId)
535  argKey = (isBinaryType ? "tb<" : "t<") + typeName + ">";
536  else
537  argKey = rpcTypeInfo.rpcTypeName;
538  if (!argLabel.isEmpty()) {
539  if (_mode.is(Mode::NamedArg) && rpcTypeInfo.typeId)
540  argKey = argLabel;
541  else if (_mode.is(Mode::NamedArg) || _mode.is(Mode::NamedTypedArg))
542  argKey = argLabel + ":" + argKey;
543  else
544  argKey += "." + argLabel;
545  }
546 
548  if (type == QMetaType::QStringList) { // Handle list
549  int i = 0;
550  QString argData;
551  if (!argDataList.isEmpty() && !argDataList.at(0).isEmpty()) {
552  argData = argDataList.at(0);
553  ++i;
554  }
555  xsm.add( argKey, argData);
556  // Output each element in list
557  for(; i < argDataList.size(); ++i) {
558  argData = argDataList.at(i);
559  if (argData.isEmpty() || argData.contains( QChar('=')))
560  xsm.add("+", argData);
561  else
562  xsm.add("", argData);
563  }
564  }
565  else {
566  xsm.add( argKey, argDataDump);
567  }
568 
569  ++nArg;
570  return true; // 1 arg added
571 }

574 void ArnRpc::pipeInput( QByteArray data)
575 {
576  if (!_receiver) {
577  errorLog( QString(tr("Can't invoke method: receiver=0")), ArnError::RpcReceiveError);
578  return;
579  }
580 
582  if (data.startsWith('"')) { // Text is received
583  int endSize = data.endsWith('"') ? (data.size() - 1) : data.size();
584  if (endSize < 1)
585  endSize = 1;
586  emit textReceived( QString::fromUtf8( data.mid( 1, endSize - 1), endSize - 1));
587  return;
588  }
589 
590  XStringMap xsmCall( data);
591 
593  QByteArray rpcFunc = xsmCall.value(0);
594  if (rpcFunc == "$heartbeat") // Built in Heart beat support
595  return funcHeartBeat( xsmCall);
596  if (rpcFunc == "$arg")
597  return funcArg( xsmCall);
598  if (rpcFunc == "$help") // Built in Help
599  return funcHelp( xsmCall);
600 
602  // qDebug() << "rpc pipeInput: data=" << data;
603  QByteArray methodName = _methodPrefix + rpcFunc;
604  if (_mode.is( Mode::UseDefaultCall)) {
605  setupReceiverMethodsParam(); // Setup searching method data structure
606 
607  int pslotIndex = _receiverMethodsParam->methodNames.indexOf( methodName);
608  if (pslotIndex < 0) { // Method not found
609  emit defaultCall( data);
610  return;
611  }
612  }
613 
615  ArgInfo argInfo[21]; // 0..9: Used args, 10..19: Default args, 20: Null arg
616  int argc = 0;
617 
618  if (_isIncludeSender) {
619  argInfo[ argc].arg = Q_ARG( ArnRpc*, this);
620  ++argc;
621  }
622 
623  bool stat = true; // Default ok
624  int index = 1; // Start after function name in xsm
625  while (index > 0) {
626  if (index >= xsmCall.size())
627  break; // End of args
628  if (argc > 10) {
629  errorLog( QString(tr("To many args:") + QString::number( argc))
630  + tr(" method=") + methodName.constData(),
631  ArnError::RpcReceiveError);
632  stat = false;
633  break;
634  }
635  stat = xsmLoadArg( xsmCall, argInfo[ argc], index, methodName);
636  if (!stat) break;
637  ++argc;
638  }
639 
640  char argOrder[10];
641  if (stat) {
642  for (int i = 0; i < 10; ++i) {
643  argOrder[i] = char(i); // Set default order 1 to 1
644  }
645  stat = argLogic( argInfo, argOrder, argc, methodName);
646  for (int i = argc; i < 10; ++i) {
647  argOrder[i] = char(20); // Set unused to null arg
648  }
649  }
650 
651  if (stat) {
652  bool useVarPar = checkConvVarPar( methodName, argc);
653  for (int i = _isIncludeSender; i < argc; ++i) {
654  stat = importArgData( argInfo[ int(argOrder[i])], methodName, useVarPar);
655  if (!stat) break;
656  }
657  }
658 
659  if (stat) {
660  if (_receiverStorage)
661  _receiverStorage->_rpcSender = this;
662  stat = QMetaObject::invokeMethod( _receiver,
663  methodName.constData(),
664  Qt::AutoConnection,
665  argInfo[ int(argOrder[0])].arg,
666  argInfo[ int(argOrder[1])].arg,
667  argInfo[ int(argOrder[2])].arg,
668  argInfo[ int(argOrder[3])].arg,
669  argInfo[ int(argOrder[4])].arg,
670  argInfo[ int(argOrder[5])].arg,
671  argInfo[ int(argOrder[6])].arg,
672  argInfo[ int(argOrder[7])].arg,
673  argInfo[ int(argOrder[8])].arg,
674  argInfo[ int(argOrder[9])].arg);
675  if (_receiverStorage)
676  _receiverStorage->_rpcSender = 0;
677  if(!stat) {
678  errorLog( QString(tr("Can't invoke method:")) + methodName.constData(),
679  ArnError::RpcReceiveError);
680  sendText("Can't invoke method, use $help");
681  }
682  }
683 
685  for (int i = _isIncludeSender; i < 20; ++i) {
686  ArgInfo& aiSlot = argInfo[i];
687  if (aiSlot.isArgAlloc && aiSlot.arg.data()) {
688  int type = QMetaType::type( aiSlot.arg.name());
689  QMetaType::destroy( type, aiSlot.arg.data());
690  }
691  if (aiSlot.isDataAlloc) {
692  void* data = const_cast<void*>( aiSlot.data);
693  QMetaType::destroy( aiSlot.typeId, data);
694  }
695  }
696 }

699 bool ArnRpc::xsmLoadArg( const XStringMap& xsm, ArgInfo& argInfo, int &index,
700  const QByteArray& methodName)
701 {
703  const QByteArray& typeKey = xsm.keyRef( index); // MW: Why ref &typeKey not working in debugger?
704  QByteArray rpcType;
705  argInfo.isPositional = true; // Default
706  int sepPos = typeKey.indexOf(':');
707  if (sepPos >=0)
708  argInfo.isPositional = false;
709  else
710  sepPos = typeKey.indexOf('.');
711  if (sepPos >=0) {
712  QByteArray t1 = typeKey.left( sepPos);
713  QByteArray t2 = typeKey.mid( sepPos + 1);
714  rpcType = argInfo.isPositional ? t1 : t2;
715  argInfo.name = argInfo.isPositional ? t2 : t1;
716  argInfo.hasType = !rpcType.isEmpty();
717  argInfo.hasName = !argInfo.name.isEmpty();
718  }
719  else {
720  rpcType = typeKey; // Assume type (can also be name)
721  }
722 
724  bool isTypeGen = false;
725  if (rpcType.startsWith("t<"))
726  isTypeGen = true;
727  else if (rpcType.startsWith("tb<")) {
728  isTypeGen = true;
729  argInfo.isBinary = true;
730  }
731 
733  const RpcTypeInfo& rpcTypeInfo = (isTypeGen || (!argInfo.hasType && _mode.is( Mode::NamedArg)))
734  ? typeInfoNull() : typeInfoFromRpc( rpcType);
735  bool isListFormat = rpcTypeInfo.typeId == QMetaType::QStringList;
736  if (isTypeGen) {
737  int posStart = rpcType.indexOf('<') + 1;
738  int posEnd = rpcType.lastIndexOf('>');
739  argInfo.qtType = rpcType.mid( posStart, (posEnd < 0 ? -1 : posEnd - posStart));
740  argInfo.typeId = QMetaType::type( argInfo.qtType.constData());
741  argInfo.hasType = true;
742  }
743  else if (rpcTypeInfo.typeId) {
744  argInfo.qtType = rpcTypeInfo.qtTypeName;
745  argInfo.typeId = rpcTypeInfo.typeId;
746  }
747  else if (rpcType.isEmpty() && argInfo.name.isEmpty()) {
748  argInfo.qtType = "QString"; // Default type;
749  argInfo.typeId = QMetaType::QString;
750  }
751  else {
752  if (!argInfo.hasType)
753  rpcType.clear(); // No type given
754  if (!argInfo.hasName) {
755  argInfo.name = typeKey; // This must be a name
756  argInfo.isPositional = false;
757  }
758  }
759 
760  bool possibleNameArg = !_mode.is( Mode::OnlyPosArgIn) && !argInfo.name.isEmpty();
761 
763  if ((!argInfo.typeId)
764  && (argInfo.hasType || argInfo.isPositional || !possibleNameArg)) {
765  errorLog( QString(tr("Unknown type:"))
766  + (argInfo.qtType.isEmpty() ? rpcType.constData() : argInfo.qtType.constData())
767  + (argInfo.hasName ? (tr(" name=") + argInfo.name.constData()) : QString())
768  + tr(" method=") + methodName.constData(),
769  ArnError::RpcReceiveError);
770  return false;
771  }
772 
774  const QByteArray& argDataDump = xsm.valueRef( index);
775  if (isListFormat) { // Handle list (QStringList)
776  QStringList* argDataList = new QStringList;
777  if (!argDataDump.isEmpty())
778  *argDataList += QString::fromUtf8( argDataDump.constData(), argDataDump.size());
779  ++index;
780  while (index < xsm.size()) {
781  const QByteArray& key = xsm.keyRef( index);
782  if ((key != "") && (key != "+"))
783  break;
784  const QByteArray& arg = xsm.valueRef( index);
785  *argDataList += QString::fromUtf8( arg.constData(), arg.size());
786  ++index;
787  }
788  argInfo.data = argDataList;
789  argInfo.dataAsArg = true;
790  argInfo.isDataAlloc = true;
791  }
792  else {
793  ++index;
794  argInfo.data = &argDataDump;
795  }
796 
797  return true;
798 }

801 bool ArnRpc::argLogic( ArgInfo* argInfo, char* argOrder, int& argc, const QByteArray& methodName)
802 {
803  if (_mode.is( Mode::OnlyPosArgIn)) return true; // Only allowed call by positional argument
804  if (_isIncludeSender) return true; // IncludeSender is deprecated and never namedArg
805 
806  bool isOnlyPositional = true;
807  bool isOnlyNamed = true;
808  for (int i = 0; i < argc; ++i) {
809  if ( argInfo[i].isPositional)
810  isOnlyNamed = false;
811  else
812  isOnlyPositional = false;
813  }
814  if (isOnlyPositional && (argc > 0))
815  return true; // Only positional arguments in call has been used
816 
817  if (!isOnlyNamed) {
818  errorLog( QString(tr("Mixed positional & named arg call not supported, method="))
819  + methodName.constData(),
820  ArnError::RpcReceiveError);
821  return false;
822  }
823 
824  int methodIndex = argLogicFindMethod( argInfo, argc, methodName);
825  if (methodIndex < 0) return false; // Failed finding a method
826 
827  const QMetaObject* metaObject = _receiver->metaObject();
828  QMetaMethod method = metaObject->method( methodIndex);
829  QList<QByteArray> parNames = method.parameterNames();
830  QList<QByteArray> parTypes = method.parameterTypes();
831  int parCount = parNames.size();
832  int defArgIndex = 10;
833  for (int parIndex = 0; parIndex < parCount; ++parIndex) {
834  bool match = false;
835  for (int argIndex = 0; argIndex < argc; ++argIndex) {
836  ArgInfo& aiSlot = argInfo[ argIndex];
837  if (aiSlot.name == parNames.at( parIndex)) { // Found arg name
838  QByteArray parType = parTypes.at( parIndex);
839  if (!aiSlot.qtType.isEmpty()) { // Type already known for arg
840  if ((parType != aiSlot.qtType)
841  && ((parType != "QVariant") || !_convVariantPar)) {
842  errorLog( QString(tr("Type mismatch, arg=")) + parNames.at( parIndex)
843  + tr(" in method=") + methodName.constData(),
844  ArnError::RpcReceiveError);
845  return false;
846  }
847  }
848  else { // Type for arg is defined by the method parameter
849  aiSlot.qtType = parType;
850  aiSlot.typeId = QMetaType::type( aiSlot.qtType.constData());
851 
852  const RpcTypeInfo& typeInfo = typeInfoFromId( aiSlot.typeId);
853  if ((aiSlot.typeId == QMetaType::QVariant) && _convVariantPar) {
854  aiSlot.qtType = "QString"; // Will be QString in QVariant
855  aiSlot.typeId = QMetaType::QString;
856  }
857  else if ((!typeInfo.typeId)
858  || typeInfo.typeId == QMetaType::QStringList) {
859  errorLog( QString(tr("Type mandatory, arg=")) + parNames.at( parIndex)
860  + tr(" type=") + aiSlot.qtType.constData()
861  + tr(" in method=") + methodName.constData(),
862  ArnError::RpcReceiveError);
863  return false;
864  }
865  }
866  argOrder[ parIndex] = char( argIndex);
867  match = true;
868  break;
869  }
870  }
871  if (!match) { // Parameter not given by arg, use default constructor
872  ArgInfo& aiSlot = argInfo[ defArgIndex];
873  aiSlot.name = parNames.at( parIndex);
874  aiSlot.qtType = parTypes.at( parIndex);
875  aiSlot.typeId = QMetaType::type( aiSlot.qtType.constData());
876 #if QT_VERSION >= 0x050000
877  aiSlot.data = QMetaType::create( aiSlot.typeId);
878 #else
879  aiSlot.data = QMetaType::construct( aiSlot.typeId);
880 #endif
881  aiSlot.isDataAlloc = true;
882  aiSlot.dataAsArg = true;
883 
884  argOrder[ parIndex] = char( defArgIndex);
885  ++defArgIndex;
886  }
887  }
888  argc = parCount; // New argc will be exactly as number of parameters
889 
890  return true;
891 }

894 int ArnRpc::argLogicFindMethod( const ArnRpc::ArgInfo* argInfo, int argc, const QByteArray& methodName)
895 {
896  setupReceiverMethodsParam(); // Setup searching method data structure
897 
898  int pslotIndex = _receiverMethodsParam->methodNames.indexOf( methodName);
899  if (pslotIndex < 0) {
900  errorLog( QString(tr("Not found, method=")) + methodName.constData(),
901  ArnError::RpcReceiveError);
902  return -1;
903  }
904 
905  const MethodsParam::Params& pslot = _receiverMethodsParam->paramTab.at( pslotIndex);
906 
907  // Only 1 method with zero parameters, use it
908  if ((pslot.paramNames.size() == 1) && pslot.paramNames.at(0).isEmpty())
909  return pslot.methodIdsTab.at(0).at(0);
910 
911  int foundArgCount = 0;
912  QList<int> methodCand = pslot.allMethodIds; // Start with all methods as candidates (same name)
913  for (int argIndex = 0; argIndex < argc; ++argIndex) {
914  int parIndex = pslot.paramNames.indexOf( argInfo[ argIndex].name);
915  if (parIndex < 0) // arg not used at all, Ok (unneeded)
916  continue;
917 
919  ++foundArgCount;
920 
922  const QList<int>& methodIds = pslot.methodIdsTab.at( parIndex);
923  for (int i = 0; i < methodCand.size();) {
924  if (methodIds.contains( methodCand.at(i))) {
925  ++i;
926  continue;
927  }
928  methodCand.removeAt(i); // Remove method not using the parameter
929  }
930  }
931 
932  if (methodCand.isEmpty()) {
933  errorLog( QString(tr("Not found method with matching parameters, method=")) + methodName.constData(),
934  ArnError::RpcReceiveError);
935  return -1;
936  }
937 
938  if (methodCand.size() == 1)
939  return methodCand.at(0); // Match with exactly 1 method
940 
942  const QMetaObject* metaObject = _receiver->metaObject();
943  for (int i = 0; i < methodCand.size();) {
944  QMetaMethod method = metaObject->method( methodCand.at(i));
945  if (method.parameterNames().size() == foundArgCount) {
946  ++i;
947  continue;
948  }
949  methodCand.removeAt(i); // Remove method with not same number of params
950  }
951 
952  if (methodCand.size() == 1)
953  return methodCand.at(0); // Match with exactly 1 method having same number of params
954 
955  errorLog( QString(tr("Many methods with matching parameters, method=")) + methodName.constData(),
956  ArnError::RpcReceiveError);
957  return -1;
958 }

961 void ArnRpc::setupReceiverMethodsParam()
962 {
963  if (_receiverMethodsParam) return; // Already done
964 
965  MethodsParam* mpar = new MethodsParam;
966  _receiverMethodsParam = mpar;
967 
968  const QMetaObject* metaObject = _receiver->metaObject();
969  int methodCount = metaObject->methodCount();
970  QByteArray lastMethodName;
971  for (int methodIndex = 0; methodIndex < methodCount; ++methodIndex) {
972  QMetaMethod method = metaObject->method( methodIndex);
973  QByteArray methodSign = methodSignature( method);
974  if (!methodSign.startsWith( _methodPrefix)) continue;
975 
977  QByteArray methodName = methodSign.left( methodSign.indexOf('('));
978  if (methodName != lastMethodName) {
979  mpar->methodNames += methodName;
980  mpar->paramTab += MethodsParam::Params();
981  lastMethodName = methodName;
982  }
983  MethodsParam::Params& pslot = mpar->paramTab[ mpar->paramTab.size() - 1];
984 
985  pslot.allMethodIds += methodIndex;
986 
987  QList<QByteArray> parNames = method.parameterNames();
988  int parCount = parNames.size();
989  int parIndexStart = parCount == 0 ? -1 : 0;
990  for (int parIndex = parIndexStart; parIndex < parCount; ++parIndex) {
991  // method with no parameters will store using parName="" (parIndex=-1)
992  const char* parName = parIndex < 0 ? "" : parNames.at( parIndex).constData();
993  int parI = pslot.paramNames.indexOf( parName);
994  if (parI < 0) {
995  pslot.paramNames += parName;
996  pslot.methodIdsTab += QList<int>();
997  parI = pslot.paramNames.size() - 1;
998  }
999  pslot.methodIdsTab[ parI] += methodIndex;
1000  }
1001  }

1348 void ArnRpc::batchConnect( const QObject *sender, const QRegExp &rgx,
1349  const QObject *receiver, const QString& replace,
1350  Mode mode)
1351 {
1352  QList<QByteArray> signalSignTab;
1353  QList<QByteArray> methodSignLowTab;
1354  if (mode.is( mode.Debug))
1355  qDebug() << "batchConnect: regExp=" << rgx.pattern() << " replace=" << replace;
1356 
1358  const QMetaObject* metaObject = sender->metaObject();
1359  QList<QByteArray> doneMethodSignTab;
1360  int methodCount = metaObject->methodCount();
1361  for (int i = 0; i < methodCount; ++i) {
1362  QMetaMethod method = metaObject->method(i);
1363  if (method.methodType() != QMetaMethod::Signal) continue; // Must be a Signal
1364 
1365  QByteArray signalSign = methodSignature( method);
1366  if (doneMethodSignTab.contains( signalSign)) continue; // Already done (inherited)
1367  doneMethodSignTab += signalSign;
1368 
1369  QString signalName = QString::fromLatin1( signalSign.left( signalSign.indexOf('(')).constData());
1370  QByteArray paramSign = signalSign.mid( signalName.size());
1371  if (rgx.indexIn( signalName) >= 0) { // Match of signal in regExp
1372  QString methodName = replace;
1373  for (int j = 1; j <= rgx.captureCount(); ++j) {
1374  methodName.replace("\\" + QString::number(j), rgx.cap(j));
1375  }
1376  signalSignTab += signalSign;
1377  QByteArray methodSignLow = (methodName.toLatin1() + paramSign).toLower();
1378  methodSignLowTab += methodSignLow;
1379  if (mode.is( mode.Debug))
1380  qDebug() << "batchConnect: try match signal=" << signalSign <<
1381  " method(low)=" << methodSignLow;
1382  }
1383  else if (mode.is( mode.Debug))
1384  qDebug() << "batchConnect: No regExp match on signal=" << signalSign;
1385  }
1386 
1388  metaObject = receiver->metaObject();
1389  methodCount = metaObject->methodCount();
1390  doneMethodSignTab.clear();
1391  QByteArray methodSignCompHead;
1392  int methodIdCompHead = -1;
1393  for (int methodId = 0; methodId < methodCount; ++methodId) {
1394  QMetaMethod method = metaObject->method(methodId);
1395  QByteArray methodSign = methodSignature( method);
1396  if (doneMethodSignTab.contains( methodSign)) continue; // Already done (inherited)
1397  doneMethodSignTab += methodSign;
1398 
1399  QByteArray methodSignComp = methodSign;
1400  methodSignComp.chop(1); // Remove last ")"
1401 
1402  if (!mode.is( Mode::NoDefaultArgs) // When using Default args ...
1403  && methodSignCompHead.startsWith( methodSignComp) // Starts with same signatur ...
1404  && hasSameParamNames( method, metaObject->method( methodIdCompHead))) // and same param names
1405  continue; // Skip it, to prohibit multiple calls.
1406 
1407  methodSignCompHead = methodSignComp;
1408  methodIdCompHead = methodId;
1409 
1410  QByteArray methodSignLow = methodSign.toLower();
1411  int index = methodSignLowTab.indexOf( methodSignLow);
1412  if (index >= 0) { // Match signal to method
1413  const char* methodPrefix = (method.methodType() == QMetaMethod::Signal) ? "2" : "1";
1414  connect( sender, "2" + signalSignTab[ index],
1415  receiver, methodPrefix + methodSign);
1416  if (mode.is( mode.Debug))
1417  qDebug() << "batchConnect: connect signal=" << signalSignTab[ index] <<
1418  " method=" << methodSign;
1419  }
1420  else if (mode.is( mode.Debug))
1421  qDebug() << "batchConnect: No match on method=" << methodSignLow;
1422  }
1423 }
